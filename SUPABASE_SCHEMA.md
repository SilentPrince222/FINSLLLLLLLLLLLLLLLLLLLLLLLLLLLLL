## 📋 Profiles Table (User Data)
```sql
create table profiles (
  id uuid references auth.users primary key,
  email text unique not null,
  role user_role default 'student',
  full_name text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Auto create profile when user signs up
create function public.handle_new_user()
returns trigger as $$
begin
  insert into public.profiles (id, email, role)
  values (new.id, new.email, (new.raw_user_meta_data->>'role')::user_role);
  return new;
end;
$$ language plpgsql security definer;

create trigger on_auth_user_created
  after insert on auth.users
  for each row execute function public.handle_new_user();
```

## ✏️ Grades Table
```sql
create table grades (
  id bigint generated by default as identity primary key,
  student_id uuid references profiles(id) on delete cascade not null,
  subject text not null,
  score integer not null check (score >= 0 and score <= 100),
  semester text not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,

  unique(student_id, subject, semester)
);
```

---

## 📅 Timetable Table
```sql
create table timetable (
  id bigint generated by default as identity primary key,
  user_id uuid references profiles(id) on delete cascade not null,
  subject text not null,
  day text not null check (day in ('Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday')),
  start_time time not null,
  end_time time not null,
  room text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);
```

---

## 📅 Events Table
```sql
create table events (
  id bigint generated by default as identity primary key,
  user_id uuid references profiles(id) on delete cascade not null,
  title text not null,
  due_date timestamp with time zone not null,
  description text,
  type text not null check (type in ('exam', 'homework', 'activity', 'holiday', 'social')),
  priority text default 'medium' check (priority in ('high', 'medium', 'low')),
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);
```

## 🔒 Row Level Security Policies

```sql
alter table profiles enable row level security;
alter table grades enable row level security;
alter table timetable enable row level security;
alter table events enable row level security;

-- Profiles: Users can view their own profile, teachers can view all students
create policy "Users view own profile" on profiles
  for select using (auth.uid() = id);

create policy "Teachers view all profiles" on profiles
  for select using (exists(
    select 1 from profiles where id = auth.uid() and role = 'teacher'
  ));

-- Grades: Students view own grades, teachers manage all grades
create policy "Students view own grades" on grades
  for select using (auth.uid() = student_id);

create policy "Teachers manage grades" on grades
  for all using (exists(
    select 1 from profiles where id = auth.uid() and role = 'teacher'
  ));

-- Timetable: Users view own timetable, teachers manage all
create policy "Users view own timetable" on timetable
  for select using (auth.uid() = user_id);

create policy "Teachers manage timetable" on timetable
  for all using (exists(
    select 1 from profiles where id = auth.uid() and role = 'teacher'
  ));

-- Events: Users view own events, teachers manage all
create policy "Users view own events" on events
  for select using (auth.uid() = user_id);

create policy "Teachers manage events" on events
  for all using (exists(
    select 1 from profiles where id = auth.uid() and role = 'teacher'
  ));
```

## ⚡ CRUD Utility Functions

```sql
-- Get student grades
create function get_student_grades(student_id uuid)
returns setof grades as $$
  select * from grades where student_id = $1 order by created_at desc;
$$ language sql stable;

-- Get student timetable
create function get_user_timetable(user_id uuid)
returns setof timetable as $$
  select * from timetable where user_id = $1 order by day, start_time;
$$ language sql stable;

-- Get user events
create function get_user_events(user_id uuid)
returns setof events as $$
  select * from events where user_id = $1 order by due_date asc;
$$ language sql stable;
```
